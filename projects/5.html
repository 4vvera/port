from earsketch import * #import all items
init() #prepares the DAW(digital audio workstation) to accept and display music
setTempo(130) # how fast the song is going to go

d = AK_UNDOG_PERC_DRUMS
t = ENTREP_THEME_OUTRO_TRUMPET
p = DKBEAR_FREE_BEAT_FULL
b = COMMON_LOVE_DRUMBEAT_1
extra = HIPHOP_BASSSUB_002
theme = BOYKINZ_NIGHT_THEME_BASS_1

def part1(beg, last):
fitMedia(p, 3, beg, last)
fitMedia(b,4,beg*2,last*3.6)
#fitMedia(clap,3,beg,last/2)
#for bar in range(beg,last):
#make a beat ( sound track,start,beat)
#makeBeat(clap, 4, bar , clapBeat)
part1(1,5)

def part2(x,y):
fitMedia(t, 2, x, y)
fitMedia(theme, 1, x*0.6, y*1.125)
fitMedia(d, 1, x*2.25, y*1.8)
part2(7,19)

clapBeat = "--0-------0--" # 0 is for the pause

def part2con(a,b):
fitMedia(ELECTRO_DRUM_MAIN_PART_014, 5, a, b*1.2)
for track in range(1,4):
#set effect (track,type,parameter , value)
setEffect(1, COMPRESSOR, COMPRESSOR_THRESHOLD, -30)
setEffect(1, COMPRESSOR, COMPRESSOR_RATIO, 100)
for bar in range(a,b):
#make a beat (sound track,start,beat)
makeBeat(extra, 6, bar , clapBeat)


part2con(13,19)

def extra(g,h):
fitMedia(TFLAMES_OMEN_BASS_VRS, 7, g, h)
extra(19,30)
